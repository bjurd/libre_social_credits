"config"
{
	// What your currency will be referred to as. It's best to keep it singular.
	"display"		"Social Credit"

	"earn"
	{
		// Strings in here must be valid numbers. Since currency is stored as whole numbers, it is only safe to use decimals
		// where you can garuntee a whole number will be produced, such as with multipliers.

		// How much players will earn per-player they kill.
		// Set to 0 or less to disable kill rewards.
		"kill_value"			"10"

		// How much extra players will earn for getting headshots.
		// Set to 0 or less to disable headshot bonuses.
		"headshot_multiplier"	"1.5"

		// How much players will lose every time they die.
		// Set to 0 or less to disable death losses.
		"death_value"			"10"

		// How much players will earn on every initial spawn.
		// A.K.A. how much every time they connect.
		// Set to 0 or less to disable.
		"join_value"			"5"
	}

	"commands"
	{
		// Values in here control how chat commands are registered and understood.
		// Note that values will be converted to lowercase and trimmed.
		//
		// Do NOT remove the key for a command to disable it, all of the keys in here need to remain or else you may have problems.
		// To disable a command, set its VALUE to an empty string (""), example:
		//
		// "pay"		""
		//
		// The prefix value is what will be need to be at the beginning of the string for the command to be recognized, so if you have:
		//
		// "prefix"		"!"
		//
		// Then in order to pay someone, you need to type "!pay" into chat. The prefix can be any length you want.
		// If it is set to an empty string (""), then all commands are disabled.
		"prefix"		"!"

		"pay"			"pay"

		"bal"			"bal"
	}
}
